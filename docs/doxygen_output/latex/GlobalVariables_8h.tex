\doxysubsection{Detailed Description}
Defines global constants and variables used in the wave propagation solver. 

This header provides physical constants, numerical parameters, and sponge-\/layer settings shared across the numerical wave model.\hypertarget{GlobalVariables_8h_autotoc_md0}{}\doxysubsubsection{\texorpdfstring{Constants}{Constants}}\label{GlobalVariables_8h_autotoc_md0}

\begin{DoxyItemize}
\item {\ttfamily G} \+: Gravitational acceleration $ [m/s^2] $
\item {\ttfamily PI} \+: Mathematical constant pi
\item {\ttfamily hmin} \+: Minimum water depth to prevent dry cells
\item {\ttfamily alpha1}, {\ttfamily gamma1} \+: Coefficients from Nwogu\textquotesingle{}s equations for dispersion optimization
\item {\ttfamily theta\+RC} \+: Reconstruction parameter in the shallow water scheme
\end{DoxyItemize}\hypertarget{GlobalVariables_8h_autotoc_md1}{}\doxysubsubsection{\texorpdfstring{Sponge Layer Parameters}{Sponge Layer Parameters}}\label{GlobalVariables_8h_autotoc_md1}

\begin{DoxyItemize}
\item {\ttfamily alpha\+SP} \+: Damping factor
\item {\ttfamily gamma\+SP} \+: Damping strength scaling factor, typically in \mbox{[}0.\+88, 0.\+92\mbox{]}
\item {\ttfamily n\+SP} \+: Number of grid points for sponge layer per boundary
\item {\ttfamily epsilon\+SP} \+: Threshold above which the sponge layer becomes inactive
\end{DoxyItemize}\hypertarget{GlobalVariables_8h_autotoc_md2}{}\doxysubsubsection{\texorpdfstring{Numerical Schemes}{Numerical Schemes}}\label{GlobalVariables_8h_autotoc_md2}
